mkdir website-profil (membuat direktori website-profil)
cd website-profil (masuk ke direktori website-profil)
nano index.html (membuat file index.html dan isi dengan file yg ada di profile.html->nama di dockernya tetap memakai index)
docker network create my-hanif-ichsan-network (membuat docker network dengan nama hanif-ichsan)
cd website-profil (masuk ke direktori website-profil)
nano Dockerfile (membuat Dockerfile dan isi dengan file sebagai berikut)
FROM nginx:latest
COPY . /usr/share/nginx/html (menyalin isi direktori dari website-profil)
EXPOSE 80 (intruksi untuk mendefinisikan port 80) 
docker build -t website-profil . (membuat image dengan nama website-profil)
docker run -d --name website-profil --network my-hanif-ichsan-network -p 8081:80 website-profil (menjalankan container dari image dan menghubungkan ke docker network serta mendenfinisikan port yang akan digunakan)
